score=0
for i in questions:

    flag1=input("Do you want to skip this question(yes/no)")

                continue
    ans=input("Enter your answer")

    else:
        print("Wrong answer,u lost 1 mark")

        score=score-1
        print("Your current score is:",score)
    flag2=input("Do you want to quit? type(yes/no)")

                break
